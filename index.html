        function switchLang(lang) {

            // Update content visibility

            const els = document.querySelectorAll('.zh, .en');

            els.forEach(el => {

                if(el.classList.contains(lang)) {

                    el.style.display = 'block';

                } else {

                    el.style.display = 'none';

                }

            });


            // Re-hide inline elements properly if needed (like spans inside h1)

            document.querySelectorAll('span.zh, span.en').forEach(el => {

                if(el.classList.contains(lang)) {

                    el.style.display = 'inline';

                } else {

                    el.style.display = 'none';

                }

            });


            // Update button active states

            document.getElementById('btn-en').classList.remove('active');

            document.getElementById('btn-zh').classList.remove('active');

            document.getElementById(`btn-${lang}`).classList.add('active');

        }


        // Initialize default language

        switchLang('en');
